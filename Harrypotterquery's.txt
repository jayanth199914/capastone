CREATE TABLE USER1234(
USER_NAME VARCHAR(40) PRIMARY KEY,
MOBILE_NUMBER INT,
EMAIL_ID VARCHAR(40),
PASSWORD VARCHAR(40));

CREATE TABLE REGISTRATION(
User_name VARCHAR2(50) PRIMARY KEY,
Password varchar2(50),
EMAIL VARCHAR2(50),
PHONE_NUMBER INT,
);

CREATE TABLE PRODUCTS (
PRODUCT_ID INT PRIMARY KEY,PRODUCT_NAME VARCHAR2(40),PRODUCT_PRICE INT,
PRODUCT_COLOR VARCHAR2(20),PRODUCT_QUANTITY INT,PRODUCT_SIZE VARCHAR2(10),
PRODUCT_DESC VARCHAR2(200));

CREATE TABLE SHIPMENT_DETAILS(
SHIPMENT_ID INT PRIMARY KEY,
SHIPMENT_DATE DATE,
ORDER_ID INT,
USER_NAME VARCHAR2(40),
CONSTRAINT FK_SP_PR FOREIGN KEY(USER_NAME) REFERENCES USER1234(USER_NAME),
CONSTRAINT FK_SP_OD FOREIGN KEY(ORDER_ID) REFERENCES ORDERDETAILS(ORDER_ID));

CREATE TABLE FEEDBACK(
USER_NAME VARCHAR2(40),
MAIL_ID VARCHAR2(20),
DESRCIPTION VARCHAR2(70),
PRODUCT_ID INT,
CONSTRAINT FK_FEDBACK_PD FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCTS(PRODUCT_ID));

CREATE TABLE COUPON(
COUPON_NAME VARCHAR2(20),
COUPON_CODE VARCHAR2(20) PRIMARY KEY,
ORIGINAL_PRICE INT,
DISCOUNT_PRICE INT,
EXPIRE_DATE DATE,
PRODUCT_ID INT,
CONSTRAINT FK_COUPON_PR FOREIGN KEY(PRODUCT_ID) REFERENCES PRODUCTS(PRODUCT_ID));


CREATE TABLE ADRRESS(
DOOR_NO INT,
STREET_NAME VARCHAR2(20),
LANDMARK VARCHAR2(20),
PINCODE INT,
STATE INT,
USER_NAME VARCHAR(40) ,
CONSTRAINT FK_AD_U FOREIGN KEY(USER_NAME) REFERENCES USER1234(USER_NAME));



CREATE TABLE PAYMENT(
PAYMENT_ID INT PRIMARY KEY,
AMOUNT INT,
PAYMENT_DATE DATE,
MODE_OF_PAY VARCHAR2(20),
ORDER_ID INT,
CONSTRAINT FK_PY_OR FOREIGN KEY(ORDER_ID) REFERENCES ORDERDETAILS(ORDER_ID));



CREATE TABLE ORDERDETAILS(
ORDER_ID INT PRIMARY KEY,
PRODUCT_ID INT,
ORDER_STATUS VARCHAR2(20),
NO_OF_ITEMS INT,
ORDER_DATE DATE,
USER_NAME VARCHAR2(40),
CONSTRAINT FK_ORDT_U FOREIGN KEY(USER_NAME) REFERENCES USER1234(USER_NAME));
